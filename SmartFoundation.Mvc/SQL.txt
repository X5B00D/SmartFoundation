CREATE VIEW [dbo].[V_GetListUserPermission]
AS
SELECT        p.permissionID, pt.permissionTypeName_A, pt.permissionTypeName_E, u.userID, m.menuName_E
FROM            dbo.Menu AS m INNER JOIN
                         dbo.MenuDistributor AS md ON md.menuID_FK = m.menuID INNER JOIN
                         dbo.Distributor AS d ON md.distributorID_FK = d.distributorID INNER JOIN
                         dbo.UserDistributor AS ud ON d.distributorID = ud.distributorID_FK INNER JOIN
                         dbo.[User] AS u ON ud.userID_FK = u.userID INNER JOIN
                         dbo.Permission AS p ON d.distributorID = p.distributorID_FK AND u.userID = p.UserID_FK INNER JOIN
                         dbo.PermissionType AS pt ON p.permissionTypeID_FK = pt.permissionTypeID
WHERE        (m.menuActive = 1) AND (md.menuDistributorActive = 1) AND (d.distributorActive = 1) AND (ud.UDActive = 1) AND (ud.UDStartDate IS NOT NULL) AND (CAST(ud.UDStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (ud.UDEndDate IS NULL) AND (u.userActive = 1) AND (p.permissionActive = 1) AND (p.permissionStartDate IS NOT NULL) AND (CAST(p.permissionStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (p.permissionEndDate IS NULL) AND (pt.permissionTypeActive = 1) OR
                         (m.menuActive = 1) AND (md.menuDistributorActive = 1) AND (d.distributorActive = 1) AND (ud.UDActive = 1) AND (ud.UDStartDate IS NOT NULL) AND (CAST(ud.UDStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (u.userActive = 1) AND (p.permissionActive = 1) AND (p.permissionStartDate IS NOT NULL) AND (CAST(p.permissionStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND (p.permissionEndDate IS NULL) AND 
                         (pt.permissionTypeActive = 1) AND (CAST(ud.UDEndDate AS date) > CAST(GETDATE() AS datetime)) OR
                         (m.menuActive = 1) AND (md.menuDistributorActive = 1) AND (d.distributorActive = 1) AND (ud.UDActive = 1) AND (ud.UDStartDate IS NOT NULL) AND (CAST(ud.UDStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (ud.UDEndDate IS NULL) AND (u.userActive = 1) AND (p.permissionActive = 1) AND (p.permissionStartDate IS NOT NULL) AND (CAST(p.permissionStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (pt.permissionTypeActive = 1) AND (CAST(p.permissionEndDate AS date) > CAST(GETDATE() AS datetime)) OR
                         (m.menuActive = 1) AND (md.menuDistributorActive = 1) AND (d.distributorActive = 1) AND (ud.UDActive = 1) AND (ud.UDStartDate IS NOT NULL) AND (CAST(ud.UDStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND 
                         (u.userActive = 1) AND (p.permissionActive = 1) AND (p.permissionStartDate IS NOT NULL) AND (CAST(p.permissionStartDate AS datetime) <= CAST(GETDATE() AS datetime)) AND (pt.permissionTypeActive = 1) AND 
                         (CAST(ud.UDEndDate AS date) > CAST(GETDATE() AS datetime)) AND (CAST(p.permissionEndDate AS date) > CAST(GETDATE() AS datetime))
GO
